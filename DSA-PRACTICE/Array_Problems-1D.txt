5..Rotate Array by K Positions
void rotate(int* nums, int numsSize, int k) {
    if(numsSize==NULL||numsSize==1||k==0)
        return;
    k=k%numsSize;
    int i=0;
    int j=numsSize-1;
    while(i<j)
    {
        int temp=nums[i];
        nums[i]=nums[j];
        nums[j]=temp;
        i++;
        j--;
    }
   i=0;
    j=numsSize-1;
    int l=k-1;
    while(i<l)
    {
        int temp=nums[i];
        nums[i]=nums[l];
        nums[l]=temp;
        i++;
        l--;
    }
    i=k;
    while(i<j)
    {
        int temp=nums[i];
        nums[i]=nums[j];
        nums[j]=temp;
        i++;
        j--;
    }
 
}

9.two sum
class Solution {
    public int[] twoSum(int[] nums, int target) {
        Map<Integer,Integer> mp = new HashMap<>();
        for(int i=0;i< nums.length ;i++){
            int complements = target - nums[i];
            if(mp.containsKey(complements)){
                return new int[] {mp.get(complements),i};
            }
            mp.put(nums[i],i);
        }
        return new int[] {};
    }
}
10.Best Time to Buy and Sell Stock
class Solution {
    public int maxProfit(int[] nums) {
        int buy = 0;
        int profit = 0;
        int max_profit = 0;
        for(int sell =0 ; sell < nums.length ; sell++){
            
            if(nums[buy] > nums[sell]){
                buy = sell;
            }else if(nums[buy] < nums[sell]){
                profit = nums[sell] - nums[buy];
            }
            max_profit = (profit>max_profit) ? profit: max_profit;
        }
        return max_profit;
    }
}
11.Product of Array Except Self
class Solution {
    public int[] productExceptSelf(int[] nums) {

        int[] arr = new int[nums.length];
        int suffix = 1;
        for(int i = 0 ; i < nums.length ; i++){
            arr[i] = suffix ; 
            suffix *= nums[i];
        }
        int prefix = 1;
        for(int  i = nums.length-1 ; i >= 0 ; i--){
            arr[i] *= prefix ;  
            prefix *= nums[i];
        }
        return arr;
    }
}
12.Longest Consecutive Sequence
class Solution {
    public int longestConsecutive(int[] nums) {
        Arrays.sort(nums);
        if(nums.length == 0)  return 0;
        int max = 0; 
        int count = 0;
        for(int i = 0 ; i < nums.length-1 ; i++){
            if(nums[i]+1==nums[i+1]){
                count++;
            }
            else if(nums[i]== nums[i+1])   continue;
            else{
                count = 0 ;
            }
            max = Math.max( count, max);
        }
        return max + 1;
    }
}

17.Trapping Rain Water
class Solution {
    public int trap(int[] height) {
        int leftm = 0 ; 
        int rightm = 0;
        int l = 0 ;
        int r = height.length-1;
        int water = 0;
        while(l<r){
            leftm = Math.max(height[l],leftm);
            rightm = Math.max(height[r],rightm);
            if(leftm < rightm){
                water += leftm - height[l];
                l++;
            } 
            else{
                water += rightm - height[r];
                r--;
            }
        }
        return water;
    }
}
18.Container With Most Water
class Solution {
    public int maxArea(int[] height) {
        int left = 0;
        int right = height.length -1 ;

        int water = 0;
        int left_max = 0 ;
        int right_max = 0 ;
        while(left < right ){
            left_max = Math.max(left_max , height[left]);
            right_max = Math.max(right_max , height[right]);

            if(left_max < right_max){
                water =  Math.max(water , left_max * (right - left));
                left++;
            }
            else{
                water =  Math.max(water , right_max * (right - left));
                right--;
            }
        }
        return water;
    }
}
19.Merge Overlapping Intervals
class Solution {
    public int[][] merge(int[][] intervals) {

        Arrays.sort(intervals ,(a,b) -> a[0] - b[0]);

        List<int[]> merged = new ArrayList<>();
        
        merged.add(intervals[0]);
        int[] prev = intervals[0]; 
        for(int i = 1 ; i < intervals.length ; i++){

            if( prev[1] >= intervals[i][0]){
                int max = Math.max(prev[1] , intervals[i][1]);
                merged.remove(merged.size()-1);
                merged.add(new int[]{ prev[0] , max});      
            }
            else{
                merged.add(intervals[i]);
            }
            prev = merged.get(merged.size()-1);
        }
        return merged.toArray(new int[merged.size()][]);

    }
}

        }

21.Intersection of Two Arrays
class Solution {
    public int[] intersection(int[] nums1, int[] nums2) {
        
        HashSet<Integer> set = new HashSet<>();
        
        Set<Integer> res_set = new HashSet<>();
        for(int i = 0 ; i< nums1.length ; i++){
            res_set.add(nums1[i]);
        }

        for(int i= 0 ; i < nums2.length ; i++){
            if(res_set.contains(nums2[i])){
                set.add(nums2[i]);
            }
        }

        int[] arr = new int[set.size()];
        int i = 0 ;
        for(int num : set){
             arr[i++] = num;
        }
        return arr; 
    }
}

27.Largest Sum Contiguous Subarray of Size K
class Solution {
    public int maxSubarraySum(int[] arr, int k) {
        
         int l = 0; 
         int r = k;
         int n = arr.length;
         int sum = 0;
         int max= 0;
         for(int i = 0 ; i < k ; i++){
             sum += arr[i];
         }
         max = sum;
         while( r < n){
            sum = sum - arr[l];
            l++;
            sum = sum + arr[r];
            r++;
            max = Math.max(sum , max);
         }
         return max;
    }
}
28. First Missing Positive
class Solution {
    public int firstMissingPositive(int[] nums) {

        for(int i = 0 ; i< nums.length ; i++){
            if(nums[i] < 0){
                nums[i] = 0 ;
            }
        }
        for(int  i = 0 ; i< nums.length ; i++){
            int val = Math.abs(nums[i])-1;
            if(val >= 0 && val <= nums.length-1){
                if(nums[val] > 0){
                    nums[val] *= -1 ; 
                }
                else if(nums[val] == 0){
                    nums[val] = -1 * (nums.length+1) ;
                }
            }
        }
        for(int  i =0 ; i < nums.length ; i++){
            if(nums[i] >= 0){
                return i+1 ;
            }
        }
    return nums.length +1;
        
    }
}
30.Find Peak Element
class Solution {
    public int findPeakElement(int[] nums) {
        int left = 0;
        int right = nums.length-1;
        int mid = 0 ;
        while (left < right) {
            mid = left+(right-left)/2;
            if(nums[mid] < nums[mid + 1]){
                left = mid + 1;
            }
            else {
                right = mid ;
            }
        }

        return left;
    }
}
